                            //--Get the values from the user (controller function)--//
                            function getValues(){
                               //--Get values from the page--//
                               let fizzValue = document.getElementById("fizzValue").value;
                               let buzzValue = document.getElementById("buzzValue").value;
                               //--Ensure values are integers (input validation)--//
                               fizzValue = parseInt(fizzValue);
                               buzzValue = parseInt(buzzValue);
                               if(Number.isInteger(fizzValue) && Number.isInteger(buzzValue)){
                                   //--Call fizzBuzz function--//
                                   let fbArray = fizzBuzz(fizzValue, buzzValue);
                                   //--Call displayNumbers function--//
                                   displayNumbers(fbArray);
                               }else{
                                   alert("You must enter integers");
                               }
                           }
                           
                           //--Evaluate function using input values (logic function)--//
                           function fizzBuzz(fizzValue, buzzValue){
                               let returnArray = [];
                               //--Get all numbers from 0 to 100--//
                               for(let i = 1; i <= 100; i++){
                                    //--Check if number is Fizz/Buzz/FizzBuzz--//
                                   if((i % fizzValue == 0) && (i % buzzValue == 0)){
                                       returnArray.push('FizzBuzz');
                                   }else if(i % fizzValue == 0){
                                       returnArray.push('Fizz');
                                   }else if(i % buzzValue == 0){
                                       returnArray.push('Buzz');
                                   }else{
                                       returnArray.push(i);
                                   }
                               }
                               return returnArray;
                           }
                           
                           //--Alternate solution without if/else statements (logic function)--//
                           function fizzBuzzB(fizzValue, buzzValue){
                               let returnArray = [];
                               //--Set default boolean logic for Fizz and Buzz--//
                               let Fizz = false;
                               let Buzz = false;
                               //--Loop through values, comparing to inputs--//
                               for(let i = 1; i <= 100; i++){
                                   Fizz = i % fizzValue == 0;
                                   Buzz = i % buzzValue == 0;
                                   //--Switch false to true if requirements are met--//
                                   switch(true){
                                       case Fizz && Buzz:{
                                           returnArray.push("FizzBuzz");
                                           break;
                                       }case Fizz:{
                                           returnArray.push("Fizz");
                                           break;
                                       }case Buzz:{
                                           returnArray.push("Buzz");
                                           break;
                                       }default:{
                                           returnArray.push(i);
                                           break;
                                       }
                                   }
                               }
                               return returnArray;
                           }
                           
                           //--Option C: Alternate solution without using '%' modulus operator (logic function)--//
                           function fizzBuzzC(fizzValue, buzzValue, arraySize){
                               let returnArray = [];
                               //--Set initial Fizz and Buzz values--//
                               let Fizz = 1;
                               let Buzz = 1;
                               //--Get all numbers from 0 to arraySize while incrementing Fizz and Buzz--//
                               for(let i = 1; i <= arraySize; i++, Fizz++, Buzz++){
                                   //--Check if number is Fizz/Buzz/FizzBuzz, then reset incrementation--//
                                  if((Fizz == fizzValue) && (Buzz == buzzValue)){
                                      returnArray.push('FizzBuzz');
                                      Fizz = 0;
                                      Buzz = 0;
                                  }else if(Fizz == fizzValue){
                                      returnArray.push('Fizz');
                                      Fizz = 0;
                                  }else if(Buzz == buzzValue){
                                      returnArray.push('Buzz');
                                      Buzz = 0;
                                  }else{
                                      returnArray.push(i);
                                  }
                              }
                               return returnArray;
                           }

                               //--Get the table body element from the page--//
                               let tableBody = document.getElementById("results");
                               //--Get the template row--//

getValues()

This is a function that was created to collect all of the relevant information from the user via the input elements contained within the form. Additionally, it will 'clean' the information coming in to make sure that it has data with the correct format, in this case we need integers. It will display the information after calling the other two functions to perform their actions.

fizzBuzz()

This function was designed to generate 100 values and determine whether or not each value qualifies as Fizz, Buzz, FizzBuzz, or none of the above. It accomplishes this using if/if else/else logic to compare each value to the remainders when dividing by the input values for Fizz and Buzz.

fizzBuzzB()

This function is an alternative solution that can be called by getValues() to solve the challenge using switch statements. By setting Fizz and Buzz variables to false by default, we can use boolean logic to switch them to true based on the same comparison to the input values as in the version above.

fizzBuzzC()

In this instance, a function ws written for a hypothetical circumstance in which the '%' modulo and '/' division operators were determined to be too resource-intensive and a solution needed to be written in such a way to use neither operator. Instead, additional variables were incremented within the For Loop and reset when triggered by the Fizz and Buzz values input by the user.

displayValues()

The final function was written to display the values in a table with 5 columns to be more efficient with vertical space. It takes advantage of the template element in the HTML page as well as CSS elements to highlight the Fizz/Buzz/FizzBuzz values. Looking at 5 values at once, this function will add the data to the "td" elements within the template while also adding the associated class to each item, which have been individually formatted.
